\section{Background}

\begin{frame}{Assertions may own resources}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.46\textwidth}
      \minorhead{Ordinary assumptions}
      \begin{itemize}
        \item reusable facts;
        \item contraction is valid;
        \item example: $x=3$.
      \end{itemize}
    \end{column}
    \begin{column}{0.49\textwidth}
      \minorhead{Spatial assumptions}
      \begin{itemize}
        \item owned resources;
        \item no spatial duplication;
        \item example: $l\mapsto v$.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.55cm}
  \keyeq{\Heap(\mathit{Loc},\mathit{Val}) = \mathit{Loc}\to\mathsf{Option}(\mathit{Val})}

  \begin{itemize}
    \item MPSL resource: an exclusive partial heap;
    \item affine logic: resources may be discarded, not duplicated.
  \end{itemize}
\end{frame}

\begin{frame}{Additive and separating conjunction}
  \begin{table}
    \centering
    \begin{tabularx}{\textwidth}{>{\bfseries}p{0.18\textwidth} Y Y}
      \toprule
      & Additive $P\land Q$ & Separating $P\Sep Q$ \\
      \midrule
      Resource
        & same heap $h$
        & disjoint heaps $h_1\disjoint h_2$ \\
      Semantics
        & $P(h)\land Q(h)$
        & $h=h_1\heapunion h_2$, $P(h_1)$, $Q(h_2)$ \\
      Proof use
        & both branches retain the context
        & spatial context is partitioned \\
      \bottomrule
    \end{tabularx}
  \end{table}

  \vspace{0.35cm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \keyeq{P\Sep Q\;\Entails\;Q\Sep P}
    \end{column}
    \begin{column}{0.49\textwidth}
      \keyeq{(l\mapsto v_1)\Sep(l\mapsto v_2)\;\Entails\;\false}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Recursive types create recursive semantic domains}
  \minorhead{List values}
  \keyeq{\mathcal V(\tau\ \mathsf{list})
    =\mu X.\;\{\mathsf{Nil}\}\cup
      \{\mathsf{Cons}(v,w)\mid
        v\in\mathcal V(\tau)\land w\in X\}}

  \begin{itemize}
    \item the definition of list values refers to the domain being defined;
    \item general recursive types induce recursive domain equations:
  \end{itemize}

  \keyeq{D\cong F(D)}

  \begin{itemize}
    \item domain theory constructs solutions under suitable conditions;
    \item step indexing instead exposes finite approximations indexed by $n$.
  \end{itemize}

  \vfill
  \muted{Motivation only: MPSL does not implement recursive types or predicates.}
\end{frame}

\begin{frame}{Step propositions are downward closed}
  \begin{itemize}
    \item \texttt{SProp}: a downward-closed predicate on $\mathbb N$;
    \item $n\in P$: $P$ is observable for $n$ steps;
    \item truth at a larger index implies truth at every smaller index.
  \end{itemize}

  \keyeq{m\le n\ \land\ n\in P\quad\Longrightarrow\quad m\in P}

  \vspace{0.25cm}
  \minorhead{Finite-observation equality}
  \keyeq{P=_nQ\quad\Longleftrightarrow\quad
    \forall m\le n,\;(m\in P\leftrightarrow m\in Q)}

\end{frame}

\begin{frame}{Later shifts the observation index}
  \keyeq{n\in\Later P\quad\Longleftrightarrow\quad
    n=0\ \lor\ \exists m,\;n=m+1\land m\in P}

  \begin{itemize}
    \item $(\Later P)_0=\true$;
    \item $(\Later P)_{n+1}=P_n$;
    \item $P\Entails\Later P$;
    \item $(\Later P\Rightarrow P)\Entails P$ \quad (L\"ob).
  \end{itemize}

  \vspace{0.35cm}
  \begin{center}
    \large $\Later P$: one weaker finite approximation of $P$.
  \end{center}

  \vfill
  \scriptsize\muted{In MPSL, $\Later$ shifts only the logical index; it is not a program step.}
\end{frame}

\begin{frame}{OFEs control higher-order observations}
  \minorhead{Ordered family of equivalences}
  \begin{itemize}
    \item $=_n$ is an equivalence relation;
    \item $n\ge m$ and $x=_n y$ imply $x=_m y$;
    \item $(\forall n,\; x =_n y)$ implies $x = y$.
  \end{itemize}

  \minorhead{Non-expansive function}
  \keyeq{x=_n y\quad\Longrightarrow\quad f(x)=_n f(y)}

  \begin{itemize}
    \item object-language arrows denote \texttt{NEFun};
    \item MPSL implements OFEs, not COFEs or contractive maps;
    \item no guarded fixed-point operator is implemented.
  \end{itemize}
\end{frame}

\section{Syntax and elaboration}

\begin{frame}{One typed syntax covers terms and propositions}
  \minorhead{Types}
  \[
    \tau ::= \mathsf{loc}\mid\mathsf{val}\mid\mathsf{iProp}\mid
    \mathbf 0\mid\mathbf 1\mid\tau\times\tau\mid\tau+\tau\mid\tau\to\tau
  \]

  \minorhead{Terms}
  \[
    \begin{aligned}
    t,u,P,Q ::= {}&x\mid()\mid\lambda\,x : \tau.\, t\mid t(u)\mid(t,u)\mid\cdots\\
      &\mid\false\mid\true\mid t =_{\tau} u\mid P\Rightarrow Q\mid P\land Q\mid P\lor Q\\
      &\mid P\Sep Q\mid P\Wand Q\mid\forall\,x : \tau.\, P\mid\exists\,x : \tau.\, P\\
      &\mid l\mapsto v\mid\Always P\mid\Later P
    \end{aligned}
  \]

  \begin{itemize}
    \item formulas are terms of type $\mathsf{iProp}$;
    \item $\mathsf{loc}$ and $\mathsf{val}$ are distinct object types.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Ty defines the object-language universe}
\begin{lstlisting}[style=leancode]
inductive Ty where
  | loc
  | val
  | iprop
  | empty
  | unit
  | prod (left right : Ty)
  | sum (left right : Ty)
  | arr (domain codomain : Ty)
\end{lstlisting}

  \begin{itemize}
    \item \texttt{loc} and \texttt{val}: distinct object-language types;
    \item \texttt{iprop}: the object type of assertions;
    \item \texttt{arr}: interpreted by non-expansive functions.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Expr constructors encode typing rules}
\begin{lstlisting}[style=leancode,basicstyle=\ttfamily\tiny]
inductive Expr (Loc Val) : List Ty -> Ty -> Type where
  | lam {ctx} {arg result} :
      Expr Loc Val (arg :: ctx) result ->
      Expr Loc Val ctx (.arr arg result)
  | app {ctx} {arg result} :
      Expr Loc Val ctx (.arr arg result) ->
      Expr Loc Val ctx arg -> Expr Loc Val ctx result
  | sep {ctx} :
      Expr Loc Val ctx .iprop -> Expr Loc Val ctx .iprop ->
      Expr Loc Val ctx .iprop
  | all {ctx} {ty} :
      Expr Loc Val (ty :: ctx) .iprop -> Expr Loc Val ctx .iprop
  | pointsTo {ctx} :
      Expr Loc Val ctx .loc -> Expr Loc Val ctx .val ->
      Expr Loc Val ctx .iprop

abbrev Formula (Loc Val) := Expr Loc Val [] .iprop
\end{lstlisting}

  \vfill
  \keyeq{\Gamma\vdash t:\tau\quad\text{is represented by}\quad
    \texttt{Expr Loc Val $\Gamma$ $\tau$}}
\end{frame}

\begin{frame}{IProp adds resources to step propositions}
  \minorhead{Assertion domain}
  \keyeq{\IProp(\mathit{Loc},\mathit{Val})
    = \left\{P : \Heap(\mathit{Loc},\mathit{Val})\to\mathsf{SProp}
      \mid P\text{ is heap-monotone}\right\}}

  \begin{itemize}
    \item $P(h)$: a downward-closed proposition over observation steps;
    \item $h_1\sqsubseteq h_2$: resource extension;
    \item heap monotonicity:
  \end{itemize}

  \keyeq{h_1\sqsubseteq h_2\ \land\ n\in P(h_1)
    \quad\Longrightarrow\quad n\in P(h_2)}

  \minorhead{Formula interpretation}
  \[
    \Sem{-} : \mathsf{Formula}(\mathit{Loc},\mathit{Val})
      \longrightarrow \IProp(\mathit{Loc},\mathit{Val}).
  \]
\end{frame}

\begin{frame}{Soundness is semantic step inclusion}
  \minorhead{Semantic entailment}
  \keyeq{P\Entails_{\!i}Q
    \quad\Longleftrightarrow\quad
    \forall h,n,\; n\in P(h)\Rightarrow n\in Q(h)}

  \minorhead{Formula judgment}
  \[
    P\Entails Q
    \quad\Longleftrightarrow\quad
    \Sem{P}\Entails_{\!i}\Sem{Q}.
  \]

  \minorhead{Soundness target}
  \keyeq{P\Entails Q
    \quad\Longrightarrow\quad
    \forall h,n,\;n\in\Sem{P}(h)\Rightarrow n\in\Sem{Q}(h)}

  \begin{itemize}
    \item proof-mode rules are theorems over \texttt{IProp.Entails};
    \item tactics compose theorem applications; Lean checks the proof term.
  \end{itemize}
\end{frame}

\begin{frame}{Why a proof mode?}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.47\textwidth}
      \minorhead{Meta-level goal}
      \[
        H_1:\phi_1,\ldots,H_n:\phi_n
        \;\vdash_{\mathrm{Lean}}\;\psi
      \]
      \begin{itemize}
        \item named hypotheses;
        \item local context transformations;
        \item \texttt{intro}, \texttt{split}, \texttt{exact}.
      \end{itemize}
    \end{column}
    \begin{column}{0.49\textwidth}
      \minorhead{Embedded entailment}
      \[
        \vdash_{\mathrm{Lean}}\;(P\Entails Q)
      \]
      \begin{itemize}
        \item $P$: one object-language term;
        \item subformulas of $P$: unnamed;
        \item Lean sees \texttt{IProp.Entails}.
      \end{itemize}
    \end{column}
  \end{columns}

  \minorhead{Proof-mode representation}
  \keyeq{\Gamma_p;\Gamma_s\vdash_{\texttt{MPSL}}Q
    \quad:=\quad\Sem{\Gamma_p;\Gamma_s}\Entails_{\!i}Q}

  \begin{itemize}
    \item names expose object-logic structure to goal-directed tactics.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Persistent and spatial proof contexts}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.51\textwidth}
\begin{lstlisting}[style=leancode]
x : Nat
hEq : x = x
|- 
hSpec : (*@$\Always$@*) `P
-------------------------- (*@$\Always$@*)
hCell : loc(0) |-> val("v")

      \begin{itemize}
        \item pure context: Lean variables and propositions;
        \item $\Gamma_p$: persistent context, reusable;
        \item $\Gamma_s$: spatial context, explicitly partitioned;
        \item names are unique across both MPSL contexts.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.25cm}
  \keyeq{\Sem{\Gamma_p;\Gamma_s}
    =\Always(P_1\land\cdots\land P_m)\Sep(Q_1\Sep\cdots\Sep Q_n)}
\end{frame}

\begin{frame}{Tactic effects: enter and decompose}
  \minorhead{\texttt{mintro h}}
  \keyeq{P\Entails Q
    \quad\Longrightarrow\quad
    \varnothing\;;\;h:P\ \vdash_{\texttt{MPSL}} Q}

  \minorhead{\texttt{mdestruct h as hP hQ} \;for spatial $P\Sep Q$}
  \keyeq{\Gamma_p\;;\;h:P\Sep Q,\Gamma_s\ \vdash_{\texttt{MPSL}} R
    \quad\Longrightarrow\quad
    \Gamma_p\;;\;h_P:P,h_Q:Q,\Gamma_s\ \vdash_{\texttt{MPSL}} R}

  \minorhead{\texttt{mpersistent h}}
  \keyeq{\Gamma_p\;;\;h:P,\Gamma_s\ \vdash_{\texttt{MPSL}} R
    \quad\Longrightarrow\quad
    \begin{cases}
      P\Entails_{\!i}\Always P,\\
      h:P,\Gamma_p\;;\;\Gamma_s\ \vdash_{\texttt{MPSL}} R.
    \end{cases}}
\end{frame}

\begin{frame}{Tactic effects: branch the goal}
  \minorhead{\texttt{msplit}}
  \keyeq{\Gamma_p\;;\;\Gamma_s\ \vdash_{\texttt{MPSL}} P\land Q
    \quad\Longrightarrow\quad
    \begin{cases}
      \Gamma_p\;;\;\Gamma_s\ \vdash_{\texttt{MPSL}} P,\\
      \Gamma_p\;;\;\Gamma_s\ \vdash_{\texttt{MPSL}} Q.
    \end{cases}}

  \minorhead{\texttt{msep [hQ]}}
  \keyeq{\Gamma_p\;;\;h_P:P,h_Q:Q\ \vdash_{\texttt{MPSL}} Q\Sep P
    \quad\Longrightarrow\quad
    \begin{cases}
      \Gamma_p\;;\;h_Q:Q\ \vdash_{\texttt{MPSL}} Q,\\
      \Gamma_p\;;\;h_P:P\ \vdash_{\texttt{MPSL}} P.
    \end{cases}}

  \begin{itemize}
    \item \texttt{msplit}: both branches keep the contexts;
    \item \texttt{msep}: persistent context copied, spatial context partitioned.
  \end{itemize}
\end{frame}

\begin{frame}{Tactic effects: close and focus}
  \minorhead{\texttt{mexact h}}
  \keyeq{\Gamma_p\;;\;h:P,\Gamma_s\ \vdash_{\texttt{MPSL}} P
    \quad\Longrightarrow\quad\checkmark}

  \minorhead{\texttt{mapply hImp hP}}
  \keyeq{\Gamma_p\;;\;h_{\mathrm{Imp}}:P\Rightarrow Q,h_P:P,\Gamma_s
    \ \vdash_{\texttt{MPSL}} Q
    \quad\Longrightarrow\quad\checkmark}

  \minorhead{\texttt{mframe hP}}
  \keyeq{\Gamma_p\;;\;h_P:P,\Gamma_s\ \vdash_{\texttt{MPSL}} P\Sep Q
    \quad\Longrightarrow\quad
    \Gamma_p\;;\;\Gamma_s\ \vdash_{\texttt{MPSL}} Q}

  \begin{itemize}
    \item other closers: \texttt{massumption}, \texttt{mtruth},
      \texttt{mfalse h}, \texttt{mrefl}.
  \end{itemize}
\end{frame}

\begin{frame}{Tactic effects: binders and later}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \minorhead{\texttt{mforall x}}
      \[
        \begin{aligned}
        &\Delta\mid\Gamma_p;\Gamma_s\vdash
          \forall\,x:\tau.\,P(x)\\
        &\quad\Longrightarrow
          \Delta,x:\Sem{\tau}\mid\Gamma_p;\Gamma_s\vdash P(x)
        \end{aligned}
      \]

      \minorhead{\texttt{mexists t}}
      \[
        \begin{aligned}
        &\Delta\mid\Gamma_p;\Gamma_s\vdash
          \exists\,x:\tau.\,P(x)\\
        &\quad\Longrightarrow
          \Delta\mid\Gamma_p;\Gamma_s\vdash P(t)
        \end{aligned}
      \]

      \minorhead{\texttt{mopenexists h as x hx}}
      \[
        \begin{aligned}
        &\Delta\mid\Gamma_p;h:\exists\,x:\tau.\,P(x),\Gamma_s\vdash Q\\
        &\quad\Longrightarrow
        \Delta,x:\Sem{\tau}\mid\Gamma_p;h_x:P(x),\Gamma_s\vdash Q
        \end{aligned}
      \]
    \end{column}
    \begin{column}{0.47\textwidth}
      \minorhead{\texttt{mlater}}
      \[
        \Gamma_p;\Gamma_s\vdash\Later P
        \Longrightarrow
        \Gamma_p;\Gamma_s\vdash P
      \]

      \minorhead{\texttt{mopenlater h as hP}}
      \[
        \begin{aligned}
        &\Gamma_p;h:\Later P,\Gamma_s\vdash\Later Q\\
        &\quad\Longrightarrow
        \Gamma_p;h_P:P,\Gamma_s\vdash Q
        \end{aligned}
      \]

      \minorhead{\texttt{mopen h as hP}}
      \[
        \begin{aligned}
        &h:\Always P,\Gamma_p;\Gamma_s\vdash Q\\
        &\quad\Longrightarrow
        h_P:P,\Gamma_p;\Gamma_s\vdash Q
        \end{aligned}
      \]
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Proof: spatial partition}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.60\textwidth}
\begin{lstlisting}[style=leancode]
example (P Q : Formula Nat String) :
    mpsl{ `P (*@$\Sep$@*) `Q } (*@$\Entails$@*)
      mpsl{ `Q (*@$\Sep$@*) `P } := by
  mintro h
  mdestruct h as hP hQ
  msep [hQ]
  (*@$\mathord{\cdot}$@*) mexact hQ
  (*@$\mathord{\cdot}$@*) mexact hP
\end{lstlisting}
    \end{column}
    \begin{column}{0.36\textwidth}
      \begin{itemize}
        \item \texttt{mdestruct}: expose $h_P$, $h_Q$;
        \item \texttt{msep [hQ]}: $h_Q$ left;
        \item remainder $h_P$ right.
      \end{itemize}
    \end{column}
  \end{columns}

  \vfill
  \muted{A spatial name cannot be assigned to both branches.}
\end{frame}

\begin{frame}[fragile]{Proof: existential}
  \[
    \bigl((\exists\,x : \mathsf{loc}.\;x=_{\mathsf{loc}}x)\Sep R\bigr)
    \Entails
    \exists\,x : \mathsf{loc}.\;R\Sep(x=_{\mathsf{loc}}x)
  \]

\begin{lstlisting}[style=leancode]
mintro h
mdestruct h as hExists hR
mopenexists hExists as x hx
mexists $! x
msep [hR]
(*@$\mathord{\cdot}$@*) mexact hR
(*@$\mathord{\cdot}$@*) mexact hx
\end{lstlisting}

  \begin{itemize}
    \item \texttt{mopenexists}: witness enters Lean's pure context;
    \item \texttt{mexists \$! x}: embed that witness back into the DSL.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Proof: persistent reuse}
\begin{lstlisting}[style=leancode]
example (P Q : Formula Nat String) :
    mpsl{ ((*@$\Always$@*) `P) (*@$\Sep$@*) `Q } (*@$\Entails$@*)
      mpsl{ `P (*@$\Sep$@*) (`P (*@$\Sep$@*) `Q) } := by
  mintro h
  mdestruct h as hBox hQ
  mpersistent hBox
  (*@$\mathord{\cdot}$@*) exact IProp.always_idem_intro P.denote
  (*@$\mathord{\cdot}$@*) mopen hBox as hP
    msep []
    (*@$\mathord{\cdot}$@*) mexact hP
    (*@$\mathord{\cdot}$@*) msep []
      (*@$\mathord{\cdot}$@*) mexact hP
      (*@$\mathord{\cdot}$@*) mexact hQ
\end{lstlisting}

  \begin{itemize}
    \item \texttt{mpersistent} requires a persistence certificate;
    \item persistent $h_P$ is available in both spatial branches.
  \end{itemize}
\end{frame}

\section{Summary}

\begin{frame}{Implemented and current scope}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \minorhead{Implemented}
      \begin{itemize}
        \item intrinsically typed higher-order DSL;
        \item exclusive partial heaps and $l\mapsto v$;
        \item affine BI, equality, and quantifiers;
        \item $\Always$, $\Later$, OFEs, and \texttt{NEFun};
        \item named contexts, proof-mode tactics, InfoView.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \minorhead{Not implemented}
      \begin{itemize}
        \item programs and operational semantics;
        \item weakest preconditions and Hoare triples;
        \item generic resources and ghost state;
        \item invariants and view shifts;
        \item recursive predicates and fixed points.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Three points to retain}
  \begin{enumerate}
    \item \textbf{Separation tracks ownership.}
      \quad $P\Sep Q$ partitions the spatial resource.
    \item \textbf{Step indexing tracks finite observation.}
      \quad \texttt{SProp}, OFEs, \texttt{NEFun}, and $\Later$.
    \item \textbf{Lean tracks the proof.}
      \quad intrinsic syntax, semantic rules, kernel checking.
  \end{enumerate}

  \vspace{0.65cm}
  \begin{center}
    \Large\textcolor{PKURed}{Higher-order, step-indexed entailments in a small checked core.}
  \end{center}
\end{frame}
